const fpcalc = require('fpcalc')
const path = require('path')

/**
 * Creates a compressed audio fingerprint of the input audio via chromaprint
 * @function fingerprint
 * @public
 * @see {@link https://acoustid.org/chromaprint|chromaprint}
 * @arg {string} inFile - Path to the input asset
 * @arg {fpCallback} cb - Callback with err and data response
 */
function main(inFile, cb) {
  fpcalc(path.resolve(inFile), function(err, result) {
    if (err) { cb(err) }
    cb(null, result.fingerprint)
  })
}
/**
 * @callback fpCallback
 * @arg {Error} err - Error object
 * @arg {string} fpData - The compressed audio fingerprint string
 */

module.exports = main
