const debug = require('debug')('larynx:spectrogram')
const path = require('path')
const { spawn } = require('child_process')
const spectrogramFlags = require('./flags.json').spectrogram

const { SOX_BIN_PATH } = require('./sox')

/**
 * Generates spectral analysis image of the input audio asset via `sox`
 * @function spectrogram
 * @public
 * @see {@link http://sox.sourceforge.net/|SoX}
 * @arg {string} file - Path to input file
 * @arg {spCallback} cb - The callback with the output image file path
 */
function main(file, cb) {
  const uri = path.resolve(file)
  const outFile = `${uri}.png`
  const soxCmd = spawn(SOX_BIN_PATH, [
    uri,
    '-n',
    'spectrogram',
    ...spectrogramFlags,
    '-t',
    path.basename(uri),
    '-o',
    outFile
  ])
  soxCmd.on('error', err => { cb(new Error(err)) })
  soxCmd.on('close', (code) => {
    debug('exitcode', code)
    cb(null, outFile)
  })
}

/**
 * @callback spCallback
 * @arg {Error} err - Error object
 * @arg {string} outFile - output image file path
 */

module.exports = main
