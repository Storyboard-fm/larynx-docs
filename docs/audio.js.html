/**
 * @public
 * @class
 * @extends nanoresource
 * @arg {string} source - The path to the source audio asset
 * @arg {Object} [opts={}] - An optional object specifying configurable options
 * @arg {Object} opts.tags - An object of tags to associate with the file
 * @arg {number|string} opts.trackNumber - Specify the `Track`'s track number
 */
class AudioFile extends Resource {
  constructor(source, opts = {}) {
    super()
    this.filename = source
    this.fd = 0
    this.type = mime.getType(this.filename)

    if (opts.tags) {
      this.tags = opts.tags
    }

    if (opts.trackNumber) {
      this.trackNumber = opts.trackNumber
    }

    try {
      if (this.type === 'audio/wav') {
        this.wav = new WaveFile()

        this.wav.fromBuffer(Buffer.from(fs.readFileSync(this.filename)))

        if (this.tags) {
          for (const tag of this.tags) {
            this.wav.setTag(...tag)
          }
        }
      }
    } catch (err) {
      console.error('Error occurred during Track instantiation! This should ' +
        'not happen. Something is very wrong.')
      throw new Error(err)
    }

  }

  /**
   * Opens the file. Once opened, work can be done on this asset
   */
  _open (cb) {
    debug('Now opening file ...', this.filename)
    fs.open(this.filename, 'r', (err, fd) => {
      if (err) return cb(err)
      this.fd = fd
      cb(null)
    })
  }

  /**
   * Closes the file. Once closed, work can no longer be performed on the asset
   */
  _close (cb) {
    debug('Now closing file ...')
    fs.close(this.fd, cb)
  }

  size (cb) {
    this.open((err) => {
      if (err) return cb(err)
      if (!this.active(cb)) return
      fs.fstat(this.fd, (err, st) => {
        if (err) return this.inactive(cb, err)
        this.inactive(cb, null, st.size)
      })
    })
  }
}
