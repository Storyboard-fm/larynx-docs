const debug = require('debug')('larynx:soxi')
const path = require('path')
const spawn = require('child_process').spawn

/**
 * @function parseSoxi
 * @private
 * @param {string} soxiStr
 * @returns {Object} - Encode metadata from the `soxi` CLI application
 */
function parseSoxi(soxiStr) {
  try {
    const soxiStats = soxiStr.split('\n').filter(f => f !== '').map(m => m.split(': ')[1])
    return {
      channels: Number(soxiStats[1]),
      sampleRate: Number(soxiStats[2]),
      bitDepth: Number(soxiStats[3].split('-bit')[0])
    }
  } catch (err) {
    console.error('Could not parse soxi output. How did that happen???')
    throw new Error(err)
  }
}

/**
 * Perform encode-level audio file introspection.
 * @function soxi
 * @public
 * @see {@link http://sox.sourceforge.net/|SoX}
 * @arg {string} file - Path to input file
 * @arg {soCallback} - Callback with the soxi data
 */
function main(file, cb) {
  let soxiStr = ''
  const soxiCmd = spawn('soxi', [path.resolve(file)])
  soxiCmd.on('error', err => { cb(err) })
  soxiCmd.stdout.on('data', d => {
    soxiStr += d
  })
  soxiCmd.on('close', code => {
    debug('exitcode', code)
    cb(null, parseSoxi(`${soxiStr}`))
  })
}

module.exports = main
