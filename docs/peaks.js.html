const debug = require('debug')('larynx:peaks')
const flags = require('./flags').audiowaveform
const fs = require('fs')
const path = require('path')
const spawn = require('child_process').spawn

/**
 * Scales the array of peaksData from `audiowaveform` to [-1...1] to meet the
 * data formatting requirements of JavaScript audio waveform modules such as
 * WaveSurfer and the BBC's peaks.js
 * @function
 * @private
 * @arg {number[]} peaksData - Array of peaks data provided by audiowaveform
 * @return {number[]} The scaled audiowaveform peaks data
 */
function scalePeaks(peaksData) {
  const maxVal = parseInt(Math.max(...[
    Math.max(...peaksData),
    Math.min(...peaksData)
  ].map(val => Math.abs(val))), 10)
  return peaksData.map(peak => peak / maxVal)
}


/**
 * Runs the `audiowaveform` CLI application on the `audioFile` in order to 
 * generate industry-compliant waveform data arrays for use in modern web UIs
 * @summary Audio peaks generator.
 * @function peaks
 * @arg {string} audioFile - Path to an audio file to analyze
 * @arg {pkCallback} cb - Callback array containing the scaled audio peaks
 */
function main(audioFile, cb) {
  spawn('audiowaveform', [
    '-i',
    path.resolve(audioFile),
    '-o',
    `${path.resolve(audioFile)}.json`,
    ...flags
  ]).on('error', err => cb(err))
    .on('close', code => {
    debug(`audiowaveform command closed with exit code ${code}`)
    const scaledPeaks = scalePeaks(
      JSON.parse(
        fs.readFileSync(`${path.resolve(audioFile)}.json`)
      ).data
    )
    cb(null, scaledPeaks)
  })
}

/**
 * @callback pkCallback
 * @arg {Error} err - Error object
 * @arg {number[]} scaledPeaks - Scaled peaks data
 */

module.exports = main
