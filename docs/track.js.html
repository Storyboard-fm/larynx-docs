/**
 * A `Track` represents a single audio asset in its entirety. It is an asset
 * intended for production purposes. This class offers interfaces to `ffmpeg`,
 * `sox`, `fpcalc`, and `audiowaveform` to produce industry-standard analyses
 * and generated assets, such as spectral imagery and audio fingerprints.
 * @public
 * @class
 * @extends AudioFile
 * @arg {string} source - The path to the source audio asset
 * @arg {Object} [opts={}] - An optional object specifying configurable options
 * @arg {Object} opts.tags - An object of tags to associate with the file
 * @arg {humber|string} opts.trackNumber - Specify the track number
 */
class Track extends AudioFile {
  constructor(source, opts = {}) {
    super(source, opts)
  }

  /**
   * Runs `fpcalc` on the audio resource, saving this fingerprint data as a
   * string in `Track.fp`. The fingerprint data is also provided via the
   * callback.
   * @see {@link https://acoustid.org/chromaprint|Chromaprint}
   * @borrows fingerprint as fingerprint
   */
  fingerprint (cb) {
    this.open((err) => {
      if (err) return cb(err)
      if (!this.active(cb)) return
      fingerprint(this.filename, (err, fp) => {
        if (err) return cb(err)
        this.fp = fp
        this.inactive(cb, null, fp)
      })
    })
  }

  /**
   * Generate audiowaveform peaks data as a scaled Array.
   * @see {@link https://github.com/bbc/audiowaveform|BBC audiowaveform}
   */
  peaks (cb) {
    this.open((err) => {
      if (err) return cb(err)
      if (!this.active(cb)) return
      peaks(this.filename, (err, pd) => {
        if (err) return cb(err)
        this.peaksData = pd
        this.inactive(cb, null, pd)
      })
    })
  }

  /**
   * Detect the audio track's silences via FFmpeg's `silencedetect` audio filter
   * @see {@link https://ffmpeg.org/ffmpeg-filters.html#silencedetect|FFmpeg Filters Documentation}
   */
  silence (cb) {
    this.open((err) => {
      if (err) return cb(err)
      const silences = []

      function validateSilences(silObj) {
        const silenceStarts = Object.keys(silObj)
        return {
          start: Number(silenceStarts.shift()) === 0,
          end: Number(silenceStarts.pop()) > 0
        }
      }

      function parseSilences(sils) {
        const starts = []
        const maps = {}
        for (const s of sils) {
          if (s.includes('silence_start: ')) {
            starts.push(parseFloat(s.split('silence_start: ')[1]))
          } else if (s.includes('silence_duration: ')) {
            maps[starts.pop()] = s.split('silence_duration: ')[1]
          }
        }
        return [ maps, validateSilences(maps) ]
      }

      const ffmpegCmd = ffmpeg(this.filename)
        .audioFilters(flags.silence)
        .format('null')
        .output('-')
        .on('stderr', d => {
          if (`${d}`.includes('silencedetect')) {
            silences.push(`${d}`)
          }
        })
        .on('error', err => { return this.inactive(cb, err) })
        .on('end', () => {
          debug('finished', silences)
          const parsedSilences = parseSilences(silences)
          this.silences = parsedSilences
          this.inactive(cb, null, parsedSilences)
        })

      ffmpegCmd.run()
    })
  }

  /**
   * Provides encoding metadata of the audio resource via `soxi`
   * @see {@link http://sox.sourceforge.net/|SoX}
   */
  soxi (cb) {
    this.open((err) => {
      if (err) return cb(err)
      if (!this.active(cb)) return
      getSoxi(this.filename, (err, so) => {
        if (err) return cb(err)
        this.format = so
        this.inactive(cb, null, so)
      })
    })
  }

  /**
   * Generates a spectrogram image of the audio resource via `sox`
   * @see {@link http://sox.sourceforge.net/|SoX}
   */
  spectrogram (cb) {
    this.open((err) => {
      if (err) return cb(err)
      if (!this.active(cb)) return
      getSpectrogram(this.filename, (err, sp) => {
        if (err) return cb(err)
        this.spectrogram = new Asset(sp, { hint: 'image' })
        this.inactive(cb, null, this.spectrogram)
      })
    })
  }

  /**
   * Perform audio introdspection on the audio resource, providing dynamics
   * information and other industry-standard measurements.
   * @see {@link http://sox.sourceforge.net/|SoX}
   */
  stats (cb) {
    this.open((err) => {
      if (err) return cb(err)
      if (!this.active(cb)) return
      getStats(this.filename, (err, st) => {
        if (err) return cb(err)
        this.stats = st
        this.inactive(cb, null, st)
      })
    })
  }

  /**
   * Generate a waveform image asset via `ffmpeg`
   * @see {@link https://ffmpeg.org/ffmpeg-filters.html#showwavespic|FFmpeg Filters Documentation}
   */
  waveform (cb) {
    this.open((err) => {
      if (err) return cb(err)
      if (!this.active(cb)) return

      const waveformPath = `${this.filename}_waveform.png`

      const ffmpegCmd = ffmpeg(this.filename)
        .complexFilter(flags.waveform)
        .output(waveformPath)
        .on('error', err => { return this.inactive(cb, err) })
        .on('end', () => {
          debug('waveform finished', waveformPath)
          this.waveform = new Asset(waveformPath, { hint: 'image' })
          this.inactive(cb, null, this.waveform)
        })

      ffmpegCmd.run()

    })
  }
}

module.exports = Track
