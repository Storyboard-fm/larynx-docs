const debug = require('debug')('larynx:stats')
const path = require('path')
const { spawn } = require('child_process')

const { SOX_BIN_PATH } = require('./sox')

function parseStats(st) {
  try {
    const stats = `${st}`.split('\n').map(stat => { return Number(stat.split(':')[1]) })
    const statsObj =  {
      duration: stats[1],
      peak: {
        db: 20 * Math.log10( stats[3] ),
        value: stats[3]
      },
      rms: {
        db: 20 * Math.log10( stats[8] ),
        value: stats[8]
      }
    }
    return statsObj
  } catch (err) {
    throw new Error(err)
  }
}

/**
 * Perform detailed audio introspection on an audio file. Provides dynamics and
 * time-domain data in particular.
 * @function stats
 * @public
 * @see {@link http://sox.sourceforge.net/|SoX}
 * @arg {string} file - Path to the input file
 * @arg {stCallback} cb - The callback with the SoX stats data
 */
function main(file, cb) {
  let statsStr = ''
  const statsObj = {}
  const soxCmd = spawn(SOX_BIN_PATH, [ path.resolve(file), '-n', 'stat' ])
  soxCmd.on('error', err => { cb(new Error(err)) })
  soxCmd.stderr.on('data', d => {
    statsStr += d
  })
  soxCmd.on('close', (code) => {
    debug('exitcode', code)
    Object.assign(statsObj, parseStats(statsStr))
    cb(null, statsObj)
  })
}

/**
 * @callback stCallback
 * @arg {Error} err - Error object
 * @arg {string} statsObj - Stats object
 */

module.exports = main
